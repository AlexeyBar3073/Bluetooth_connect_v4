1. Где именно парсинг строк

Это прямо внутри btTransportTask.

🔻 Фрагмент:
static char lineBuffer[2500];
static size_t linePos = 0;
while (rbPop(c) && processed < (int)parseLimit) {

    if (c == '\r') continue;

    if (c == '\n') {
        if (linePos > 0) {
            lineBuffer[linePos] = '\0';

            char* start = lineBuffer;
            while (*start == ' ' || *start == '\t') start++;

            if (*start) {
                dr.publishString(TOPIC_MSG_INCOMING, start);
            }

            linePos = 0;
        }
    } else {
        if (linePos < sizeof(lineBuffer) - 1) {
            lineBuffer[linePos++] = (char)c;
        } else {
            linePos = 0;
            DBG_PRINTLN("[BT RX] Line buffer overflow");
        }
    }
}

✔ Что здесь происходит (по факту)

Это не просто прием байтов:

2. Где именно OTA-оптимизация

Она не оформлена как отдельный режим, но в коде есть явные OTA-ориентированные решения.

🔻 1. Комментарий разработчика
// Оптимизации для OTA:
//   - Watermark контроль (30%/70%/90%) — backpressure при переполнении
//   - Адаптивный размер парсинга (512-4096) — быстрее при больших данных
//   - TX-очередь 2048 байт — пиковые нагрузки OTA-чанков
//   - Динамическая задержка — 0 при >70% заполнения

🔻 2. Адаптивный chunk (под большие данные)
#define PARSE_CHUNK_MIN  512
#define PARSE_CHUNK_MAX  4096

inline size_t getAdaptiveParseChunk() {
    size_t used = rbUsed();
    if (used > RX_WATERMARK_CRIT) return PARSE_CHUNK_MAX;
    if (used > RX_WATERMARK_HIGH)  return PARSE_CHUNK_MAX / 2;
    return PARSE_CHUNK_MIN;
}

👉 Это явно сделано под большие потоки данных (OTA)

🔻 3. Увеличенная TX очередь
txQueue = xQueueCreate(1, 2048);

👉 2KB строка — нетипично для команд, но типично для OTA чанков

🔻 4. Динамическая задержка
vTaskDelay(rbUsed() > RX_WATERMARK_HIGH ? 0 : 1);

👉 при большой загрузке:

задача не спит → максимальная пропускная способность
✔ Вывод

OTA не выделена как режим, но:

вся логика (chunk, watermark, delay) явно под нее заточена

Весь протокол обмена между этим приложением и другим основан исключительно на формате JSON. То есть при работе в основном режиме команды приходят в json пакета х и ответы уходят в виде json пакетов.
При работе в режиме OTA формат обмена сохраняется. чанки приходят в составе пакета json, в котором кроме самого чанка передается еще и его номер, чтобы блок ota ориентировался нужно ли его писать и за какой блок отчитываться, так же в составе json пакета приходит и контрольная сумма чтобы модуль json мог сверить данные чанка после получения.
Все пакеты друг от друга разделяют \n, чтобы он не слипались и чтобы было легче отделить один пакет от другого, не затрачивая при этом дополнительных усилий по анализу данных.

Тогда давай зафиксируем: в твоей системе нет бинарного OTA, у тебя единый текстовый протокол (JSON + \n) для всех режимов. Это важное уточнение — и оно меняет оценку.

Я предлагаю создать такую модель BT (возможно даже объектную), при котором она в обычном режиме работала с меньшими локальными буферами, а при переходе на режим OTA (по команде старта, нужно подписаться на команды) запускаем вторую задачу приема данных, у нее будет большой кольцевой буфер, так как память уже освободилась к этому моменту (перед созданием буфера читать кучу, когда куча позволит создать большой буфер, создаем его). Backpressure сделать таким, чтобы он выбрасывал целое сообщение, а не его часть. В кольцевом буфере должны быть только цельные сообщения, куски никому не нужны. В итоге мы будет иметь одну стандартную задачу, которая в обычном режиме занимается приемом маленьких пакетов и заниматься отправкой, и вторую задачу, которая будет активироваться когда переходим в режим OTA, она будет управлять большими буферами.

/*
================================================================================
BT OTA TRANSPORT MODULE
================================================================================

Назначение:
-----------
Данный модуль реализует транспортный уровень для OTA-режима поверх Bluetooth.
Работает исключительно в OTA-режиме и предназначен для высокопроизводительного
приема JSON-сообщений, содержащих OTA-чанки.

Ключевые особенности:
---------------------
1. Работа ТОЛЬКО в OTA режиме (одноразовый runtime)
2. Отсутствие Router / Protocol слоев
3. Работа напрямую с OTA сервисом через очередь сообщений
4. Гарантия целостности сообщений:
   - В буфер попадают ТОЛЬКО завершенные сообщения (разделитель '\n')
   - Частичные сообщения НИКОГДА не передаются дальше
5. Backpressure:
   - При переполнении буфера отбрасывается ВСЁ текущее сообщение
   - Поток синхронизируется по следующему '\n'
6. Динамическое управление памятью:
   - Размер буфера определяется во время старта на основе heap

Архитектура:
------------
[ Bluetooth ]
      ↓
[ Byte Stream ]
      ↓
[ Framer (в этом модуле) ]
      ↓
[ Queue<Message> ]
      ↓
[ OTA Service ]

Принцип работы:
---------------
1. Чтение данных из SerialBT
2. Побайтовая обработка
3. Сборка сообщения до '\n'
4. Проверка:
   - не превышен ли размер
   - не было ли overflow
5. Если всё OK → отправка в OTA очередь
6. Если ошибка → discard до следующего '\n'

Важно:
------
- Модуль НЕ парсит JSON
- Модуль НЕ знает ничего о структуре OTA
- Только транспорт + framing

================================================================================
*/

#include <Arduino.h>
#include "BluetoothSerial.h"
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "freertos/queue.h"
#include "esp_heap_caps.h"

extern BluetoothSerial SerialBT;

// ================= CONFIG =================

#define OTA_MAX_MESSAGE_SIZE   4096   // Максимальный JSON пакет
#define OTA_QUEUE_LENGTH       4      // Очередь сообщений

// ================= MESSAGE STRUCT =================

struct OtaMessage {
    uint16_t len;
    char data[OTA_MAX_MESSAGE_SIZE];
};

// ================= GLOBALS =================

static QueueHandle_t otaQueue = nullptr;
static TaskHandle_t btTaskHandle = nullptr;

// ================= INTERNAL STATE =================

static char* currentBuffer = nullptr;
static size_t currentPos = 0;
static bool overflow = false;

// ================= HELPER =================

static void discardUntilNewline()
{
    while (SerialBT.available()) {
        char c = SerialBT.read();
        if (c == '\n') {
            break;
        }
    }
}

// ================= MAIN TASK =================

static void btOtaTask(void* param)
{
    const size_t READ_CHUNK = 256;
    uint8_t temp[READ_CHUNK];

    while (true)
    {
        int available = SerialBT.available();

        if (available > 0)
        {
            int toRead = min(available, (int)READ_CHUNK);
            int readBytes = SerialBT.readBytes((char*)temp, toRead);

            for (int i = 0; i < readBytes; i++)
            {
                char c = temp[i];

                if (c == '\r') continue;

                if (c == '\n')
                {
                    if (!overflow && currentPos > 0)
                    {
                        currentBuffer[currentPos] = '\0';

                        OtaMessage msg;
                        msg.len = currentPos;
                        memcpy(msg.data, currentBuffer, currentPos + 1);

                        xQueueSend(otaQueue, &msg, 0);
                    }

                    // reset state
                    currentPos = 0;
                    overflow = false;
                }
                else
                {
                    if (!overflow)
                    {
                        if (currentPos < OTA_MAX_MESSAGE_SIZE - 1)
                        {
                            currentBuffer[currentPos++] = c;
                        }
                        else
                        {
                            // overflow detected
                            overflow = true;
                        }
                    }
                }
            }
        }

        vTaskDelay(1);
    }
}

// ================= PUBLIC API =================

bool btOtaTransportStart()
{
    // выделяем буфер под текущее сообщение
    currentBuffer = (char*)heap_caps_malloc(OTA_MAX_MESSAGE_SIZE, MALLOC_CAP_INTERNAL);
    if (!currentBuffer) return false;

    otaQueue = xQueueCreate(OTA_QUEUE_LENGTH, sizeof(OtaMessage));
    if (!otaQueue) return false;

    xTaskCreatePinnedToCore(
        btOtaTask,
        "BT_OTA_TASK",
        4096,
        nullptr,
        5,
        &btTaskHandle,
        0
    );

    return true;
}

QueueHandle_t btOtaGetQueue()
{
    return otaQueue;
}

{
    QueueHandle_t queue = btOtaGetQueue();
    OtaMessage msg;

    while (true)
    {
        if (xQueueReceive(queue, &msg, portMAX_DELAY))
        {
            StaticJsonDocument<4096> doc;

            DeserializationError err = deserializeJson(doc, msg.data);
            if (err)
            {
                continue; // битый JSON
            }

            const char* cmd = doc["cmd"];
            if (!cmd || strcmp(cmd, "ota") != 0)
            {
                continue;
            }

            int chunkId = doc["chunk"];
            const char* data = doc["data"];
            uint32_t crc = doc["crc"];

            if (!data) continue;

            size_t dataLen = strlen(data);

            if (!validateCrc(data, crc))
            {
                continue;
            }

            writeChunk(chunkId, (const uint8_t*)data, dataLen);
        }
    }
}

/*
================================================================================
BT OTA TRANSPORT MODULE (ADAPTIVE BUFFER VERSION)
================================================================================

Назначение:
-----------
Высокопроизводительный транспорт для OTA по Bluetooth с адаптивным управлением
памятью. Модуль автоматически подбирает размер буферов в зависимости от
доступной heap-памяти на момент старта OTA.

Ключевые особенности:
---------------------
1. Адаптивный размер буфера:
   - минимум: 8 KB
   - максимум: 32 KB
   - определяется по largest free block (а не просто free heap!)
2. Гарантия целостности сообщений:
   - только завершённые JSON пакеты (\n)
   - частичные сообщения отбрасываются
3. Backpressure:
   - drop whole message (никогда не частично)
4. Полное отсутствие глобальных статических буферов
5. Минимум копирований

Алгоритм выбора размера буфера:
-------------------------------
1. Получаем:
   - общий свободный heap
   - largest allocatable block
2. Используем только largest block (защита от fragmentation)
3. Берём не более 70% от largest block
4. Ограничиваем диапазоном [8KB ... 32KB]

Почему так:
-----------
- ESP32 может иметь много свободной памяти, но маленькие contiguous блоки
- malloc(32KB) может упасть при freeHeap=100KB
- поэтому критичен именно largest_free_block

Архитектура:
------------
[SerialBT]
    ↓
[Framer (этот модуль)]
    ↓
[Queue<Message>]
    ↓
[OTA Service]

================================================================================
*/

#include <Arduino.h>
#include "BluetoothSerial.h"
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "freertos/queue.h"
#include "esp_heap_caps.h"

extern BluetoothSerial SerialBT;

// ================= CONFIG =================

#define OTA_MIN_BUFFER   (8 * 1024)
#define OTA_MAX_BUFFER   (32 * 1024)
#define OTA_QUEUE_LENGTH 4

// ================= MESSAGE =================

struct OtaMessage {
    uint16_t len;
    char* data;   // динамический указатель
};

// ================= GLOBAL =================

static QueueHandle_t otaQueue = nullptr;
static TaskHandle_t btTaskHandle = nullptr;

// ================= INTERNAL =================

static char* msgBuffer = nullptr;
static size_t msgBufferSize = 0;

static size_t currentPos = 0;
static bool overflow = false;

// ================= BUFFER SELECTION =================

static size_t selectOptimalBufferSize()
{
    size_t freeHeap = heap_caps_get_free_size(MALLOC_CAP_INTERNAL);
    size_t largestBlock = heap_caps_get_largest_free_block(MALLOC_CAP_INTERNAL);

    Serial.printf("[OTA] Free heap: %u\n", freeHeap);
    Serial.printf("[OTA] Largest block: %u\n", largestBlock);

    if (largestBlock < OTA_MIN_BUFFER)
    {
        Serial.println("[OTA] ERROR: not enough contiguous memory");
        return 0;
    }

    size_t target = largestBlock * 7 / 10; // 70%

    if (target > OTA_MAX_BUFFER)
        target = OTA_MAX_BUFFER;

    if (target < OTA_MIN_BUFFER)
        target = OTA_MIN_BUFFER;

    Serial.printf("[OTA] Selected buffer: %u\n", target);

    return target;
}

// ================= TASK =================

static void btOtaTask(void* param)
{
    const size_t READ_CHUNK = 256;
    uint8_t temp[READ_CHUNK];

    while (true)
    {
        int available = SerialBT.available();

        if (available > 0)
        {
            int toRead = min(available, (int)READ_CHUNK);
            int readBytes = SerialBT.readBytes((char*)temp, toRead);

            for (int i = 0; i < readBytes; i++)
            {
                char c = temp[i];

                if (c == '\r') continue;

                if (c == '\n')
                {
                    if (!overflow && currentPos > 0)
                    {
                        // создаём сообщение РОВНО нужного размера
                        OtaMessage msg;
                        msg.len = currentPos;

                        msg.data = (char*)heap_caps_malloc(currentPos + 1, MALLOC_CAP_INTERNAL);
                        if (msg.data)
                        {
                            memcpy(msg.data, msgBuffer, currentPos);
                            msg.data[currentPos] = '\0';

                            xQueueSend(otaQueue, &msg, 0);
                        }
                    }

                    currentPos = 0;
                    overflow = false;
                }
                else
                {
                    if (!overflow)
                    {
                        if (currentPos < msgBufferSize - 1)
                        {
                            msgBuffer[currentPos++] = c;
                        }
                        else
                        {
                            // переполнение → дальше игнорируем до \n
                            overflow = true;
                        }
                    }
                }
            }
        }

        vTaskDelay(1);
    }
}

// ================= PUBLIC =================

bool btOtaTransportStart()
{
    size_t bufferSize = selectOptimalBufferSize();
    if (bufferSize == 0) return false;

    msgBuffer = (char*)heap_caps_malloc(bufferSize, MALLOC_CAP_INTERNAL);
    if (!msgBuffer)
    {
        Serial.println("[OTA] Failed to allocate msgBuffer");
        return false;
    }

    msgBufferSize = bufferSize;

    otaQueue = xQueueCreate(OTA_QUEUE_LENGTH, sizeof(OtaMessage));
    if (!otaQueue)
    {
        Serial.println("[OTA] Failed to create queue");
        return false;
    }

    xTaskCreatePinnedToCore(
        btOtaTask,
        "BT_OTA_TASK",
        4096,
        nullptr,
        5,
        &btTaskHandle,
        0
    );

    Serial.println("[OTA] Transport started");

    return true;
}

QueueHandle_t btOtaGetQueue()
{
    return otaQueue;
}

/*
================================================================================
BT OTA TRANSPORT V3 (ADAPTIVE + FRAMING)
================================================================================

Назначение:
-----------
Приём потока байтов из Bluetooth (SerialBT), выделение целых сообщений по '\n'
и передача их в очередь OTA сервиса.

Свойства:
---------
- Адаптивный буфер (8–32KB) по largest free block
- Только целые сообщения (JSON строки)
- Backpressure: при overflow — discard до '\n'
- НИКАКОГО парсинга JSON
- Всегда вычитывает BT (не допускает переполнения стека BT)

Поток:
------
SerialBT → [Framer] → Queue<OtaMessage> → OTA Service

================================================================================
*/

#include <Arduino.h>
#include "BluetoothSerial.h"
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "freertos/queue.h"
#include "esp_heap_caps.h"

extern BluetoothSerial SerialBT;

// ================= CONFIG =================

#define OTA_MIN_BUFFER   (8 * 1024)
#define OTA_MAX_BUFFER   (32 * 1024)
#define OTA_QUEUE_LEN    4
#define READ_CHUNK       256

// ================= MESSAGE =================

struct OtaMessage {
    uint16_t len;
    char* data; // malloc'd, освобождается в OTA сервисе
};

// ================= STATE =================

static QueueHandle_t s_queue = nullptr;
static TaskHandle_t  s_task  = nullptr;

static char*  s_buf = nullptr;   // буфер текущего сообщения
static size_t s_cap = 0;         // размер буфера
static size_t s_pos = 0;         // текущая длина
static bool   s_ovf = false;     // overflow флаг

// ================= HELPERS =================

static size_t selectBufferSize()
{
    size_t freeHeap    = heap_caps_get_free_size(MALLOC_CAP_INTERNAL);
    size_t largest     = heap_caps_get_largest_free_block(MALLOC_CAP_INTERNAL);

    Serial.printf("[BT OTA] free=%u largest=%u\n", freeHeap, largest);

    if (largest < OTA_MIN_BUFFER) return 0;

    size_t target = (largest * 7) / 10; // 70%
    if (target > OTA_MAX_BUFFER) target = OTA_MAX_BUFFER;
    if (target < OTA_MIN_BUFFER) target = OTA_MIN_BUFFER;

    Serial.printf("[BT OTA] buf=%u\n", target);
    return target;
}

// ================= TASK =================

static void btTask(void*)
{
    uint8_t tmp[READ_CHUNK];

    while (true)
    {
        int avail = SerialBT.available();
        if (avail > 0)
        {
            int toRead = min(avail, (int)READ_CHUNK);
            int n = SerialBT.readBytes((char*)tmp, toRead);

            for (int i = 0; i < n; ++i)
            {
                char c = (char)tmp[i];
                if (c == '\r') continue;

                if (c == '\n')
                {
                    if (!s_ovf && s_pos > 0)
                    {
                        OtaMessage m;
                        m.len  = s_pos;
                        m.data = (char*)heap_caps_malloc(s_pos + 1, MALLOC_CAP_INTERNAL);
                        if (m.data)
                        {
                            memcpy(m.data, s_buf, s_pos);
                            m.data[s_pos] = '\0';
                            xQueueSend(s_queue, &m, 0);
                        }
                    }
                    // reset
                    s_pos = 0;
                    s_ovf = false;
                }
                else
                {
                    if (!s_ovf)
                    {
                        if (s_pos < s_cap - 1)
                        {
                            s_buf[s_pos++] = c;
                        }
                        else
                        {
                            // переполнение текущего сообщения
                            s_ovf = true;
                        }
                    }
                }
            }
        }

        // важно: не блокироваться надолго — BT нужно постоянно вычитывать
        vTaskDelay(1);
    }
}

// ================= PUBLIC =================

bool btOtaTransportStart()
{
    size_t cap = selectBufferSize();
    if (cap == 0) return false;

    s_buf = (char*)heap_caps_malloc(cap, MALLOC_CAP_INTERNAL);
    if (!s_buf) return false;
    s_cap = cap;

    s_queue = xQueueCreate(OTA_QUEUE_LEN, sizeof(OtaMessage));
    if (!s_queue) return false;

    xTaskCreatePinnedToCore(
        btTask, "BT_OTA_V3", 4096, nullptr, 5, &s_task, 0);

    Serial.println("[BT OTA] started");
    return true;
}

QueueHandle_t btOtaGetQueue()
{
    return s_queue;
}

        {
            StaticJsonDocument<4096> doc;
            auto err = deserializeJson(doc, msg.data);

            if (!err)
            {
                const char* cmd = doc["cmd"];
                if (cmd && strcmp(cmd, "ota") == 0)
                {
                    const char* data = doc["data"];
                    uint32_t crc = doc["crc"];
                    bool last = doc["last"] | false;

                    if (data && crcOk(data, crc))
                    {
                        size_t n = decodeBase64(data, bin, sizeof(bin));
                        if (n > 0)
                        {
                            esp_ota_write(s_ota, bin, n);
                        }

                        if (last)
                        {
                            esp_ota_end(s_ota);
                            esp_ota_set_boot_partition(s_part);
                            s_done = true;
                        }
                    }
                }
            }

            // ВАЖНО: освобождаем память сообщения
            free(msg.data);
        }
    }
}

void otaServiceStart()
{
    xTaskCreatePinnedToCore(
        otaTask, "OTA_V3", 8192, nullptr, 5, &s_task, 1);
}

void otaServiceWaitAndReboot()
{
    while (!s_done)
    {
        vTaskDelay(10);
    }

    Serial.println("[OTA] DONE → reboot");
    delay(100);
    esp_restart();
}
